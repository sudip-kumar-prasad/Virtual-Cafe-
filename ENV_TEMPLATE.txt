# Environment Variables Template
# Copy these values to your deployment platforms

## Backend Environment Variables (Render)
NODE_ENV=production
PORT=10000
DB_HOST=your_planetscale_host_here
DB_USER=your_planetscale_username_here
DB_PASSWORD=your_planetscale_password_here
DB_NAME=virtual-cafe
JWT_SECRET=your_super_secret_jwt_key_make_it_long_and_random_at_least_32_characters

REACT_APP_API_URL=https://your-backend-name.onrender.com/api
